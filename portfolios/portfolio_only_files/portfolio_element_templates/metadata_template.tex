    \subsection*{Articles included}
    % \expandafter\detokenize\expandafter{\articleList}\\ % all in line
    \begin{wrtxListMeta}
        %%% START WITH EMPTY ITEM
        \hiddenitem % EMPTY ITEM NECESSARY TO AVOID ERROR
        \newcounter{localPartCount}
        \newcounter{localArtCount}
        \foreach \article [count=\n] in \articleList {
            \ifthenelse{\equal{\article}{}}%%% make sure value is not empty
            {}% if empty, ignore
            {% if not empty
                \checkIsPart{\article}
                {%is part
                    \stepcounter{localPartCount}
                    \item[{\wrtxListLabelStyle{\textbullet\ PART\ \thelocalPartCount}[wrtxColorPrimary]}]%
                    \textcolor{wrtxColorPrimary}
                    {%
                        \hyperlink{part:\thelocalPartCount}
                        {%
                            \parsePartString{\article}%
                        }%
                        \hrule%
                    }%
                }
                {% is article
                    \stepcounter{localArtCount}
                    \renewcommand{\wrtxarticleKeyCore}{\article}
                    \item[\wrtxListLabelStyle{Chapter\ \thelocalArtCount}] \hyperlink{wrtxarticle:\article:\wrtxLanguage:articleHeader}{
                        \expandafter\detokenize\expandafter{\article}\ :%
                        \expandafter\wrtxCiteEntry\expandafter{wrtxarticle:\article:\wrtxLanguage}{title}
                    }
                }
            }
        }
    \end{wrtxListMeta}%
